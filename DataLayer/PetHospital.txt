enum Gender{
	Male, Female, Unknown
}

user{
	id	string;
	name	string;
	passwd	string;
	phoneNumber	string;
	email	string;
	gender	Gender;
	token  string;//only for access control purpose
}

//only for admin purpose
UserGroup{
	group map<user, string>;
}

Test{
	testID	int;
	testName	string;
	testDescription	string;
	startTime	datetime;
	endTime	datetime;
	duration	datedelta;
	Problems	map<Problem, point>;
	confidential	boolean;
	status		string;//即将开始、 正在进行、已结束
	creator	user;
}

Choice{
	descrptio	string;
	isAnswer	boolean;//发送给前端时全部为NULL
}

Problem{
	ID	int;
	problem	string;//题目,可以是html文本
	choices	list<Choice>;//选项, 可以是html文本

	creator	user;
}

Answer{
	answerID	int;
	problemID	int;
	userID		int;
	choiceID	int;
}

TestParticipation{
	user	User;
	test	Test;
	beginTime	datetime;
	answers	list<Answer>; //default NULL
	score	float;
}
